- [ ] **Step 1: Write entity CRUD integration tests**

Test Project, Workspace, Repo CRUD via real `SqlEntityStorage` + `StorageRepositoryAdapter` with in-memory SQLite (`PdoDatabase::createSqlite()`).

```php
<?php

declare(strict_types=1);

namespace Claudriel\Tests\Integration\Entity;

use Claudriel\Entity\Project;
use Claudriel\Entity\Repo;
use Claudriel\Entity\Workspace;
use PHPUnit\Framework\Attributes\CoversClass;
use PHPUnit\Framework\Attributes\Test;
use PHPUnit\Framework\TestCase;
use Waaseyaa\Entity\EntityType;
use Waaseyaa\Entity\Storage\Sql\SqlEntityStorage;
use Waaseyaa\Entity\Storage\StorageRepositoryAdapter;
use Waaseyaa\Foundation\Database\PdoDatabase;
use Symfony\Component\EventDispatcher\EventDispatcher;

#[CoversClass(Project::class)]
#[CoversClass(Workspace::class)]
#[CoversClass(Repo::class)]
final class ProjectWorkspaceRepoIntegrationTest extends TestCase
{
    private PdoDatabase $db;

    protected function setUp(): void
    {
        $this->db = PdoDatabase::createSqlite();
    }

    #[Test]
    public function can_crud_project(): void
    {
        $repo = $this->makeRepo('project', Project::class, ['id' => 'prid', 'uuid' => 'uuid', 'label' => 'name'], [
            'prid' => ['type' => 'integer', 'readOnly' => true],
            'uuid' => ['type' => 'string', 'readOnly' => true],
            'name' => ['type' => 'string', 'required' => true],
            'description' => ['type' => 'string'],
            'status' => ['type' => 'string'],
            'account_id' => ['type' => 'string'],
            'tenant_id' => ['type' => 'string'],
            'created_at' => ['type' => 'timestamp', 'readOnly' => true],
            'updated_at' => ['type' => 'timestamp', 'readOnly' => true],
        ]);

        // Create
        $project = new Project(['name' => 'Test Project', 'status' => 'active', 'tenant_id' => 't1']);
        $repo->save($project);
        self::assertNotNull($project->id());

        // Read
        $loaded = $repo->load($project->id());
        self::assertSame('Test Project', $loaded->get('name'));

        // Update
        $loaded->set('name', 'Updated Project');
        $repo->save($loaded);
        $reloaded = $repo->load($loaded->id());
        self::assertSame('Updated Project', $reloaded->get('name'));

        // Delete
        $repo->delete($loaded);
        self::assertNull($repo->load($loaded->id()));
    }

    // Similar tests for Workspace and Repo...

    private function makeRepo(string $id, string $class, array $keys, array $fields): StorageRepositoryAdapter
    {
        $entityType = new EntityType(id: $id, label: ucfirst($id), class: $class, keys: $keys, fieldDefinitions: $fields);
        $storage = new SqlEntityStorage($entityType, $this->db, new EventDispatcher());
        return new StorageRepositoryAdapter($storage);
    }
}
```
